import React, { useState } from 'react';
import './App.css';
import StudentForm from './StudentForm';
import TabPanel from './TabPanel';

function App() {
  const [currentTab, setCurrentTab] = useState(0);
  const tabs = ['Student A', 'Student B', 'Student C'];

  return (
    <div className="container">
      <h1 className="title">Student Forms</h1>

      <div className="tab-buttons">
        {tabs.map((label, index) => (
          <button
            key={index}
            className={`tab-button ${currentTab === index ? 'active' : ''}`}
            onClick={() => setCurrentTab(index)}
          >
            {label}
          </button>
        ))}
      </div>

      <div className="tab-content">
        <TabPanel isActive={currentTab === 0}>
          <StudentForm studentName="Student A" />
        </TabPanel>
        <TabPanel isActive={currentTab === 1}>
          <StudentForm studentName="Student B" />
        </TabPanel>
        <TabPanel isActive={currentTab === 2}>
          <StudentForm studentName="Student C" />
        </TabPanel>
      </div>
    </div>
  );
}

export default App;



import React from 'react';
import './TabPanel.css';

function TabPanel({ children, isActive }) {
  return (
    <div className={`tab-panel ${isActive ? 'visible' : 'hidden'}`}>
      {children}
    </div>
  );
}

export default TabPanel;








import React, { useState } from 'react';
import './StudentForm.css';

function StudentForm({ studentName }) {
  const [form, setForm] = useState({ name: '', age: '', grade: '' });

  const handleChange = (e) => {
    const { name, value } = e.target;
    setForm((prev) => ({ ...prev, [name]: value }));
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    alert(`Submitted for ${studentName}:\n` + JSON.stringify(form, null, 2));
  };

  return (
    <form className="form" onSubmit={handleSubmit}>
      <h2>{studentName}</h2>
      <div className="form-group">
        <label>Name</label>
        <input name="name" value={form.name} onChange={handleChange} placeholder="Enter name" />
      </div>
      <div className="form-group">
        <label>Age</label>
        <input name="age" type="number" value={form.age} onChange={handleChange} placeholder="Enter age" />
      </div>
      <div className="form-group">
        <label>Grade</label>
        <input name="grade" value={form.grade} onChange={handleChange} placeholder="Enter grade" />
      </div>
      <button type="submit" className="submit-button">Submit</button>
    </form>
  );
}

export default StudentForm;
